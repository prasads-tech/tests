
      - name: Set up GitHub CLI
        uses: actions/setup-gh@v2
        with:
          version: 'latest'

      - name: Commit and Push changes
        run: |
          git config user.name "github-actions"
          git config user.email "actions@github.com"
          git checkout -b feature/automated-updates
          git add path/to/output_directory
          git commit -m "Add generated files"
          git push origin feature/automated-updates

      - name: Create Pull Request
        run: |
          gh pr create --base main --head feature/automated-updates \
                       --title "Automated PR: Update generated files" \
                       --body "This PR includes the latest generated files from the workflow."
        env:
          GITHUB_TOKEN: ${{ secrets.GITHUB_TOKEN }}
